```java
public class Main {

	public static void main(String[] args) {
		
		final double PI = 3.14159;
				
		//PI = 4; 
		// Above line of code will give error.
		// Anything that is declared as 'final' cannot be changed or updated later in the program.
		// It is common practise to write the name of any final variable in all-uppercase letters like the name 'PI' above.
		System.out.println(PI);
		
	}
}
```
