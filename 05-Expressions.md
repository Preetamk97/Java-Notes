```java
public class Main {
	
	public static void main(String[] args) {
		
		// Expression = operand + operator
		// operand = values, variables, numbers, quantities
		// operator = + - * / % etc.
		
		int friends = 10;
		
		// Arithmetic Operators
		//**********************
		
		friends = friends + 1; // 11
		friends = friends - 1; // 10
		friends = friends * 2; // 20
		friends = friends / 5; // 4
		friends = friends % 2; // 0
		
		// Increment - Decrement Operators
		//**********************************
		
		friends =  5;
		
		friends++ ; // friends = friends + 1 = 6  // Increment Operator
		
		friends-- ; // friends = friends - 1 = 5  // Decrement Operator
		
		System.out.println(friends); // 5
	}
}
```
